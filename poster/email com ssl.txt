Envio de E-Mail utilizando SSL.

Exemplos de Configurações:

[ Gmail ]

Sender=e-mail remetente

OutFile=e-mail destinatario

SmtpServer=smtp.gmail.com

User=usuario ex: user@gmail.com

AUTENTICACAO=necessário ou não autenticar no servidor ex: TRUE, FALSE

Password=senha

SSL=TRUE

SmtpPort=465


[ Yahoo ]

Sender=e-mail remetente

OutFile=e-mail destinatário

SmtpServer=smtp.mail.yahoo.com.br

User=usuario ex: user@yahoo.com.br

Password=senha

SSL=TRUE

AUTENTICACAO=necessário ou não autenticar no servidor ex: TRUE, FALSE

SmtpPort=465



[ Hotmail ]


Sender=e-mail remetente

OutFile=e-mail destinatário

SmtpServer=smtp.live.com

User=usuario ex: user@hotmail.com

Password=senha

SSL=TRUE

AUTENTICACAO=necessário ou não autenticar no servidor ex: TRUE, FALSE

SmtpPort=587
